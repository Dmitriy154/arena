# 1. Создание проекта (если начинаете новую папку)
npm create vite@latest . -- --template vanilla
# 2. Установка зависимостей (после создания или скачивания проекта)
npm install
# 3. Установка Phaser (если его нет в проекте)
npm install phaser

ctrl + c        остановить сервер разработки
Ctrl + `        терминал

npm run dev     запуск игры (локального сервера)
npm run build   сборка

r+enter         рестарт сервера
